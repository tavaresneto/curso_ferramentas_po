15/// Representa uma instância do problema de sequenciamento de tarefas
16#[derive(Debug)]
17struct Problem {
18    /// Número de tarefas
19    n: usize,
20    /// Tempo de processamento de cada tarefa
21    processing_time: Vec<f64>,
22    /// Peso de cada tarefa
23    weight: Vec<f64>,
24    /// Data de vencimento de cada tarefa
25    due_date: Vec<f64>,
26}
27
28impl Problem {
29    /// Construtor para criar uma nova instância do problema
30    /// com os parâmetros fornecidos
31    /// Retorna uma nova instância do problema
32    /// # Arguments
33    /// * `n` - Número de tarefas
34    /// * `processing_time` - Vetor com o tempo de processamento de cada tarefa
35    /// * `weight` - Vetor com o peso de cada tarefa
36    /// * `due_date` - Vetor com a data de vencimento de cada tarefa
37    /// # Returns
38    /// * `Self` - Nova instância do problema
39    fn new(n: usize, processing_time: Vec<f64>, weight: Vec<f64>, due_date: Vec<f64>) -> Self {
40        Self { n, processing_time, weight, due_date }
41    }
42
43    /// Função para obter um problema de brinquedo (toy problem)
44    /// Retorna uma nova instância do problema de brinquedo
45    /// # Returns
46    /// * `Self` - Nova instância do problema de brinquedo
47    /// Dados:
48    /// | Tarefa | Tempo de Processamento | Peso | Data de Vencimento |
49    /// |--------|-------------------------|------|--------------------|
50    /// | 0      | 0.0                     | 0.0  | 0.0                |
51    /// | 1      | 2.0                     | 3.0  | 5.0                |
52    /// | 2      | 4.0                     | 1.0  | 3.0                |
53    /// | 3      | 3.0                     | 4.0  | 6.0                |
54    /// | 4      | 5.0                     | 2.0  | 4.0                |
55    /// | 5      | 1.0                     | 5.0  | 7.0                |
56    #[allow(dead_code)]
57    fn get_toy_problem() -> Self {
58        let n = 6;
59        let processing_time = vec![0.0, 2.0, 4.0, 3.0, 5.0, 1.0];
60        let weight = vec![0.0, 3.0, 1.0, 4.0, 2.0, 5.0];
61        let due_date = vec![0.0, 5.0, 3.0, 6.0, 4.0, 7.0];
62        Self::new(n, processing_time, weight, due_date)
63        }
64
65    /// Função para carregar uma instância do problema a partir de um arquivo
66    /// # Arguments
67    /// * `filename` - Nome do arquivo contendo a instância do problema
68    /// # Returns
69    /// * `Self` - Nova instância do problema carregada do arquivo
70    fn load_from_file(filename: &str) -> Self {
71        let file = File::open(Path::new(filename)).expect(format!("Não foi possível abrir o arquivo: {}", filename).as_str());
72        let reader = BufReader::new(file);
73        let mut lines = reader.lines();
74
75        let n_jobs = lines.next()
76            .unwrap()
77            .unwrap()
78            .parse::<usize>()
79            .unwrap();
80
81        let mut processing_times:Vec<f64> = Vec::with_capacity(n_jobs);
82        let mut weights:Vec<f64> = Vec::with_capacity(n_jobs);
83        let mut due_dates:Vec<f64> = Vec::with_capacity(n_jobs);
84
85        for (i, line) in lines.enumerate() {
86            match line {
87                Ok(linha_val) => {
88                    let colunas = linha_val
89                        .trim() //&str
90                        .split_whitespace() //SplitWhiteSpace<'_>
91                        .into_iter() //IntoIter<&str>
92                        .map(|x| x.parse::<f64>().unwrap())
93                        .collect::<Vec<f64>>();
94                    processing_times.push(colunas[0]); // Processing time
95                    due_dates.push(colunas[1]); // Due date
96                    weights.push(colunas[2]); // Weight
97                },
98                Err(_) => break,
99            }
100        }
101
102        Problem::new(n_jobs, processing_times, weights, due_dates)
103    }
104}

106/// Representa a configuração do algoritmo ACO

155/// Representa uma formiga no algoritmo ACO
156/// com os atributos necessários para sua operação
157/// e tomada de decisões
158/// # Parameters
159/// * `aco_config` - Referência à configuração do ACO
160/// * `problem` - Referência à instância do problema
161/// * `tabu_list` - Lista tabu para rastrear tarefas já visitadas
162/// * `curr_pos` - Posição atual da formiga
163/// * `tour` - Sequência de tarefas visitadas pela formiga
164struct Ant<'a> {
165    aco_config: &'a ACOConfig,
166    problem: &'a Problem,
167    tabu_list: Vec<bool>,
168    curr_pos: usize,
169    tour: Vec<usize>
170}
171
172impl<'a> Ant<'a> {
173    /// Construtor para criar uma nova formiga
174    /// com os parâmetros fornecidos
175    /// Retorna uma nova formiga
176    /// # Arguments
177    /// * `aco_config` - Referência à configuração do ACO
178    /// * `problem` - Referência à instância do problema
179    /// * `start_pos` - Posição inicial da formiga
180    /// # Returns
181    /// * `Self` - Nova formiga posicionada na tarefa inicial
182    fn init(aco_config: &'a ACOConfig, problem: &'a Problem, start_pos: usize) -> Self {
183        let mut tabu_list = vec![false; problem.n];
184        tabu_list[start_pos] = true;
185        Self {
186            aco_config,
187            problem,
188            tabu_list,
189            curr_pos: start_pos,
190            tour: vec![start_pos],
191        }
192    }
193
194    /// Move a formiga para a próxima posição
195    /// * Atualiza a posição atual da formiga
196    /// * Marca a posição como visitada na lista tabu
197    /// * Adiciona a posição ao tour da formiga
198    /// 
199    /// # Arguments
200    /// * `next_pos` - Próxima posição para a qual a formiga deve se mover  
201    fn move_to(&mut self, next_pos: usize) {
202        self.curr_pos = next_pos;
203        self.tabu_list[next_pos] = true;
204        self.tour.push(next_pos);
205    }
206
207    /// Escolhe a próxima posição para a formiga se mover
208    /// com base na matriz de feromônio e nas regras do ACO
209    /// # Arguments
210    /// * `pheromone_matrix` - Matriz de feromônio utilizada para a decisão
211    /// # Returns
212    /// * `Option<usize>` - Próxima posição escolhida ou None se não houver
213    fn choose_next_position(&self, pheromone_matrix: &mut Vec<Vec<f64>>) -> Option<usize> {
214        let mut possib:Vec<f64> = Vec::with_capacity(self.problem.n);
215        let mut sum = 0.0;
216        for j in 0..self.problem.n {
217            if !self.tabu_list[j] {
218                let visibility = 1.0 / self.problem.processing_time[j];
219                let pheromone = pheromone_matrix[self.curr_pos][j];
220                let val = pheromone.powf(self.aco_config.alpha) * visibility.powf(self.aco_config.beta);
221                possib.push(val);
222                sum += val;
223            } else {
224                possib.push(0.0);
225            }
226        }
227        if sum == 0.0 {
228            return None; 
229        }
230        let mut rng = rand::rng();
231        let mut sorteio:f64 = sum * rng.random_range(0.0..=1.0);
232        let mut curr_id = 0;
233        sorteio -= possib[curr_id];
234        while sorteio > 0.0 {
235            curr_id += 1;
236            sorteio -= possib[curr_id];
237        }
238        Some(curr_id)
239    }
240}

242/// Atualiza a matriz de feromônio com base nas soluções encontradas pelas formigas
243/// # Arguments
244/// * `pheromone_matrix` - Matriz de feromônio a ser atualizada
245/// * `ants` - Vetor de formigas que contribuíram para a atualização
246fn update_pheromone_matrix(pheromone_matrix: &mut Vec<Vec<f64>>, ants: &Vec<Ant>) {
247    for i in 0..pheromone_matrix.len() {
248        for j in 0..pheromone_matrix.len() {
249            pheromone_matrix[i][j] *= ants[0].aco_config.evaporation_rate; // Evaporation
250        }
251    }
252
253    for ant in ants {
254        let contribution = ant.aco_config.pheromone_add; // Example contribution
255        for k in 0..ant.tour.len() - 1 {
256            let from = ant.tour[k];
257            let to = ant.tour[k + 1];
258            pheromone_matrix[from][to] += contribution;
259        }
260    }
261}

263/// Função principal do algoritmo ACO
264/// # Arguments
265/// * `problem` - Referência à instância do problema
266/// * `config` - Referência à configuração do ACO
267/// # Returns
268/// * `(f64, Vec<usize>)` - Melhor fitness encontrado e o tour correspondente
269fn aco_main(problem: &Problem, config: &ACOConfig) -> (f64, Vec<usize>){
270    let mut ants:Vec<Ant> = Vec::with_capacity(config.n_ants);
271    let mut pheromone_matrix = vec![vec![1.0; problem.n]; problem.n];
272
273    let mut best_tour:Vec<usize> = Vec::new();
274    let mut best_fitness = f64::INFINITY;
275
276    for _ in 0..config.n_ants {
277        ants.push(Ant::init(
278            config, 
279            problem, 
280            0
281        ));
282    }
283
284    for _ in 0..config.n_iterations {
285        for ant in ants.iter_mut() {
286            for _ in 0..problem.n - 1 {
287                let next_pos = ant.choose_next_position(&mut pheromone_matrix);
288                match next_pos {
289                    Some(pos) => ant.move_to(pos),
290                    None => break,
291                }
292            }
293            let fitness = get_fitness(problem, &ant.tour);
294            if fitness < best_fitness {
295                best_fitness = fitness;
296                best_tour = ant.tour.clone();
297            }
298        }
299        update_pheromone_matrix(&mut pheromone_matrix, &ants);
300    }
301    (best_fitness, best_tour)
302}

304/// Calcula o fitness de um tour dado uma instância do problema
305/// # Arguments
306/// * `problem` - Referência à instância do problema
307/// * `tour` - Vetor representando o tour a ser avaliado
308/// # Returns
309/// * `f64` - Valor do fitness (atraso total ponderado) calculado para o tour
310fn get_fitness(problem: &Problem, tour: &Vec<usize>) -> f64 {
311    let mut time = 0.0;
312    let mut total_tardiness = 0.0;
313    for &job in tour {
314        time += problem.processing_time[job];
315        if time > problem.due_date[job] {
316            let tardiness = time - problem.due_date[job];
317            total_tardiness += problem.weight[job] * tardiness;
318        }
319    }
320    total_tardiness
321}
